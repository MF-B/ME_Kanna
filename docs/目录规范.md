# 目录规范
```
ME_Kanna/
├── .minecraft/          # 游戏资源 (mods, icon-exports-x32 等)
├── docs/                # 架构与契约文档
├── collector/           # 采集端 Lua 代码
│   ├── main.lua         #   主计算机入口
│   ├── meter.lua        #   海龟产能监测入口
│   ├── startup.lua      #   主计算机 OTA 启动脚本
│   ├── startup_meter.lua#   海龟 OTA 启动脚本
│   └── lib/             #   共享模块
├── backend/             # Go 后端代码
│   ├── main.go          #   程序入口
│   └── internal/        #   内部包
│       ├── api/         #     HTTP & WebSocket 路由
│       ├── model/       #     数据结构定义
│       ├── service/     #     业务逻辑
│       ├── store/       #     内存状态管理
│       └── utils/       #     工具函数 (icon, name 解析等)
└── frontend/            # 前端代码
    └── src/
        ├── api/         #   HTTP 请求封装
        ├── components/  #   UI 组件
        ├── composables/ #   组合式函数
        ├── stores/      #   状态管理 (Static + Dynamic)
        ├── locales/     #   i18n 语言文件
        └── assets/      #   静态资源
```
